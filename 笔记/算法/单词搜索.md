给定一个 m x n 二维字符网格 board 和一个字符串单词 word 。如果 word 存在于网格中，返回 true ；否则，返回 false 。

单词必须按照字母顺序，通过相邻的单元格内的字母构成，其中“相邻”单元格是那些水平相邻或垂直相邻的单元格。同一个单元格内的字母不允许被重复使用。



使用回溯法，依次在board中寻找和单词第一个字符相等的位置（i，j），从该位置进入backtrack，分别向该位置的四个方向上拓展，如果四个方向上的字符等于word中的第二个字符，递归进入回溯，直至匹配全部word字符。

在回溯过程中，因为同一个单元格内的字母不允许重复，要使用一个used数组保存访问过的节点信息。

我的解答：

```c++
class Solution {
public:
    bool exist(vector<vector<char>>& board, string word) {
        vector<vector<bool>> used(board.size(), vector<bool>(board[0].size(), false));

        for (int i = 0; i < board.size(); i++) {
            for (int j = 0; j < board[0].size(); j++) {
                if (board[i][j] == word[0]) {
                    used[i][j] = true;
                    bool ret = backtrack(1, i, j, board, word, used);
                    if (ret) {
                        return ret;
                    }
                    used[i][j] = false;
                }
            }
        }

        return false;
    }

    bool backtrack(int cur, int l, int c, vector<vector<char>>& board, string word, vector<vector<bool>>& used) {
        if (cur == word.size()) {
            return true;
        }

        bool ret = false;

        if (c < board[0].size()-1 && !used[l][c+1] && word[cur] == board[l][c+1]) {
            used[l][c+1] = true;
            ret = backtrack(cur+1, l, c+1, board, word, used);
            if (ret) return ret;
            used[l][c+1] = false;
        }

        if (l < board.size()-1 && !used[l+1][c] && word[cur] == board[l+1][c]) {
            used[l+1][c] = true;
            ret = backtrack(cur+1, l+1, c, board, word, used);
            if (ret) return ret;
            used[l+1][c] = false;
        }

        if (l > 0 && !used[l-1][c] && word[cur] == board[l-1][c]) {
            used[l-1][c] = true;
            ret = backtrack(cur+1, l-1, c, board, word, used);
            if (ret) return ret;
            used[l-1][c] = false;
        }

        if (c > 0 && !used[l][c-1] && word[cur] == board[l][c-1]) {
            used[l][c-1] = true;
            ret = backtrack(cur+1, l, c-1, board, word, used);
            if (ret) return ret;
            used[l][c-1] = false;
        }

        return false;
    }
};
```



上述解答有大量重复相似的代码，下面是官方解答：

```c++
class Solution {
public:
    bool check(vector<vector<char>>& board, vector<vector<int>>& visited, int i, int j, string& s, int k) {
        if (board[i][j] != s[k]) {
            return false;
        } else if (k == s.length() - 1) {
            return true;
        }
        visited[i][j] = true;
        vector<pair<int, int>> directions{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        bool result = false;
        for (const auto& dir: directions) {
            int newi = i + dir.first, newj = j + dir.second;
            if (newi >= 0 && newi < board.size() && newj >= 0 && newj < board[0].size()) {
                if (!visited[newi][newj]) {
                    bool flag = check(board, visited, newi, newj, s, k + 1);
                    if (flag) {
                        result = true;
                        break;
                    }
                }
            }
        }
        visited[i][j] = false;
        return result;
    }

    bool exist(vector<vector<char>>& board, string word) {
        int h = board.size(), w = board[0].size();
        vector<vector<int>> visited(h, vector<int>(w));
        for (int i = 0; i < h; i++) {
            for (int j = 0; j < w; j++) {
                bool flag = check(board, visited, i, j, word, 0);
                if (flag) {
                    return true;
                }
            }
        }
        return false;
    }
};
```
